def splitByCamNo(df):
    rowIndex=0
    eachCamNo_Rows = dict()
    for _, values in df['フルパス'].items():
        infos=values.split('\\')
        camNo=infos[len(infos)-4]
        if camNo in eachCamNo_Rows:
            eachCamNo_Rows[camNo].append(rowIndex)
        else :
            eachCamNo_Rows[camNo]=[rowIndex]
        #increment row index
        rowIndex+=1
    return eachCamNo_Rows
